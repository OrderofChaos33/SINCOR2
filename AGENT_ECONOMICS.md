```python
def calculate_commission(deal_amount, agent_id, touch_type):
    base_rate = commission_rates[touch_type]  # 3%, 2%, 2%, 10%

    # Get agent's marketplace adjustments
    marketplace = AgentMarketplace()
    adjustments = marketplace.calculate_agent_commission(agent_id, deal_amount, touch_type)

    # Final = (base_rate * tier_multiplier) + boost_percent
    final_commission = adjustments['commission']

    return final_commission, adjustments
```
